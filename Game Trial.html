    <p>Answer the following questions with 'yes' or 'no'.</p>


    <button onclick="askQuestion('Is the birthday celebrant a fan of chocolate cake?')">Ask Question 1</button>

    <button onclick="askQuestion('Do they enjoy outdoor activities?')">Ask Question 2</button>

    <button onclick="askQuestion('Do they have a favorite color that\'s not a primary color?')">Ask Question 3</button>


    <script>

        function askQuestion(question) {

            var answer = prompt(question + " (yes/no)").toLowerCase();

            if (answer === 'yes') {

                alert("That's great!");

            } else if (answer === 'no') {

                alert("Oh, I see.");

            } else {

                alert("Please enter 'yes' or 'no'.");

            }

        }

    </script>
